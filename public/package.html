    3)multidplyr

    install.packages("multidplyr")
    library(multidplyr)

    # Create a local data frame
    df <- data.frame(x = 1:10, y = 11:20)

    # Convert to a distributed data frame
    ddf <- df %>% as.tbl_lazy() %>% partition(x)

    # Perform parallel operations
    result <- ddf %>%
    group_by(x) %>%
    mutate(z = x + y)

    # Collect the result
    collect(result)
